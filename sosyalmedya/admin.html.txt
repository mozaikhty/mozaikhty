        const app = initializeApp(firebaseConfig); const auth = getAuth(app); const db = getFirestore(app);

        const ADMIN_USERNAME = "realdincer"; 
        
        let myUsername = null;

                myUsername = user.displayName || user.email.split('@')[0];

        document.getElementById('tab-users-btn').addEventListener('click', () => {
            document.getElementById('users-section').style.display = 'block';
            document.getElementById('tickets-section').style.display = 'none';
            document.getElementById('tab-users-btn').classList.add('active');

        });

            document.getElementById('users-section').style.display = 'none';

            document.getElementById('tab-tickets-btn').classList.add('active');
            document.getElementById('tab-users-btn').classList.remove('active');
        });

        // 1. KULLANICILARI ÇEK VE YÖNET
        function loadUsers() {
            onSnapshot(collection(db, "users"), (snapshot) => {
                const list = document.getElementById('users-list');
                list.innerHTML = '';
                
                snapshot.forEach(docSnap => {
                    const username = docSnap.id;
                    const data = docSnap.data();
                    
                    const isVerified = data.isVerified || false;
                    const isBanned = data.isBanned || false;

                    let badgesHtml = '';
                    if (isVerified) badgesHtml += '<span class="badge bg-blue">VIP (Mavi Tik)</span>';
                    if (isBanned) badgesHtml += '<span class="badge bg-red">BANLI</span>';

                    list.innerHTML += `
                        <div class="list-item">
                            <div class="user-info">
                                @${username} ${badgesHtml}
                            </div>
                            <div class="action-btns">
                                <button class="btn-verify" onclick="window.toggleVerify('${username}', ${isVerified})">
                                    ${isVerified ? 'Mavi Tiki Al' : 'Mavi Tik Ver'}
                                </button>
                                <button class="${isBanned ? 'btn-unban' : 'btn-ban'}" onclick="window.toggleBan('${username}', ${isBanned})">
                                    ${isBanned ? 'Banı Kaldır' : 'Sistemden Banla'}
                                </button>
                            </div>
                        </div>
                    `;
                });
            });
        }

        // 2. GELEN TALEPLERİ ÇEK (TICKETS)
        function loadTickets() {
            const q = query(collection(db, "tickets"), orderBy("createdAt", "desc"));
            onSnapshot(q, (snapshot) => {
                const list = document.getElementById('tickets-list');
                list.innerHTML = '';
                
                if(snapshot.empty) { list.innerHTML = '<p>Şu an hiçbir destek talebi yok.</p>'; return; }

                snapshot.forEach(docSnap => {
                    const ticketId = docSnap.id;
                    const data = docSnap.data();
                    
                    let dateStr = "Tarih Yok";
                    if(data.createdAt) {
                        const d = data.createdAt.toDate();
                        dateStr = `${d.getDate()}/${d.getMonth()+1}/${d.getFullYear()} - ${d.getHours().toString().padStart(2, '0')}:${d.getMinutes().toString().padStart(2, '0')}`;
                    }

                    list.innerHTML += `
                        <div class="ticket-item">
                            <div class="ticket-header">
                                <span>Gönderen: <b>@${data.sender}</b> | Konu: @${data.reportedProfile} Profili</span>
                                <span>${dateStr}</span>
                            </div>
                            <div class="ticket-msg">"${data.message}"</div>
                            <button class="delete-ticket" onclick="window.deleteTicket('${ticketId}')">Sil (Kapat)</button>
                            <div style="clear:both;"></div>
                        </div>
                    `;
                });
            });
        }

        // GLOBALE EKLENEN AKSİYON FONKSİYONLARI
        window.toggleVerify = async function(username, currentStatus) {
            if(confirm(`@${username} kişisinin Mavi Tik durumunu değiştirmek istiyor musun?`)) {
                await updateDoc(doc(db, "users", username), { isVerified: !currentStatus });
            }
        };

        window.toggleBan = async function(username, currentStatus) {
            let msg = currentStatus ? `@${username} kullanıcısının BAN'ını kaldırıyorsun. Emin misin?` : `DİKKAT: @${username} kalıcı olarak sistemden atılacak. Emin misin?`;
            if(confirm(msg)) {
                await updateDoc(doc(db, "users", username), { isBanned: !currentStatus });
            }
        };

        window.deleteTicket = async function(ticketId) {
            if(confirm("Bu talebi çözüldü olarak işaretleyip silmek istiyor musun?")) {
                await deleteDoc(doc(db, "tickets", ticketId));
            }
        };
